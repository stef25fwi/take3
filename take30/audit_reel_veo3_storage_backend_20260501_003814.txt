    31	class CloudFunctionsVeoVideoGenerationService implements VeoVideoGenerationService {
    32	  CloudFunctionsVeoVideoGenerationService({
    33	    VeoSceneGenerationService? service,
    34	    this.pollAttempts = 12,
    35	    this.pollDelay = const Duration(seconds: 2),
    36	  }) : _service = service ?? VeoSceneGenerationService();
    37	
    38	  final VeoSceneGenerationService _service;
    39	  final int pollAttempts;
    40	  final Duration pollDelay;
    41	
    42	  @override
    43	  Future<AiGeneratedVideo> generateSceneIntroVideo({
    44	    required String sceneDraftId,
    45	    required String prompt,
    46	    int durationSeconds = 15,
    47	    String aspectRatio = '16:9',
    48	  }) async {
    49	    try {
    50	      var job = await _service.requestVeoScenePreview(
    51	        sceneId: sceneDraftId,
    52	        prompt: prompt,
    53	        durationSeconds: durationSeconds,
    54	        aspectRatio: aspectRatio,
    55	      );
    56	
    57	      for (var attempt = 0; attempt < pollAttempts; attempt++) {
    58	        if (job.isCompleted) {
    59	          return _toAiGeneratedVideo(job, prompt, durationSeconds, aspectRatio);
    60	        }
    61	        if (job.isFailed) {
    62	          throw VeoServiceException(
    63	            job.errorMessage ?? 'Le backend VEO a retourné un échec.',
    64	          );
    65	        }
    66	        await Future<void>.delayed(pollDelay);
    67	        job = await _service.checkVeoSceneGeneration(sceneId: sceneDraftId);
    68	      }
    69	
    70	      throw VeoServiceException(
    71	        'La génération VEO prend plus de temps que prévu.',
    72	      );
    73	    } on VeoSceneGenerationException catch (error) {
    74	      throw VeoServiceException(
    75	        error.message,
    76	      );
    77	    }
    78	  }
    79	
    80	  static AiGeneratedVideo _toAiGeneratedVideo(
    81	    VeoGenerationJob job,
    82	    String prompt,
    83	    int durationSeconds,
    84	    String aspectRatio,
    85	  ) {
    86	    final generatedAt = job.updatedAt ?? DateTime.now();
    87	    return AiGeneratedVideo(
    88	      provider: 'veo3',
    89	      prompt: job.prompt.isEmpty ? prompt : job.prompt,
    90	      videoUrl: job.videoUrl ?? '',
    91	      thumbnailUrl: job.thumbnailUrl,
    92	      durationSeconds: job.durationSeconds,
    93	      aspectRatio: job.aspectRatio.isEmpty ? aspectRatio : job.aspectRatio,
    94	      status: aiIntroVideoStatusFromString(job.status.value),
    95	      generatedAt: generatedAt,
    96	      updatedAt: generatedAt,
    97	    );
    98	  }
    99	}

    18	class VeoSceneGenerationService {
    19	  VeoSceneGenerationService({FirebaseFunctions? functions})
    20	    : _functions = functions;
    21	
    22	  final FirebaseFunctions? _functions;
    23	
    24	  FirebaseFunctions get _resolvedFunctions =>
    25	    _functions ?? FirebaseFunctions.instanceFor(region: 'europe-west1');
    26	
    27	  Future<VeoGenerationJob> requestVeoScenePreview({
    28	    required String sceneId,
    29	    required String prompt,
    30	    int durationSeconds = 15,
    31	    String aspectRatio = '16:9',
    32	  }) async {
    33	    try {
    34	      final callable =
    35	          _resolvedFunctions.httpsCallable('startVeoSceneGeneration');
    36	      final result = await callable.call(<String, dynamic>{
    37	        'sceneId': sceneId,
    38	        'prompt': prompt,
    39	        'durationSeconds': durationSeconds,
    40	        'aspectRatio': aspectRatio,
    41	      });
    42	      return VeoGenerationJob.fromJson(
    43	        _asMap(result.data),
    44	        fallbackSceneId: sceneId,
    45	        fallbackPrompt: prompt,
    46	        fallbackDurationSeconds: durationSeconds,
    47	        fallbackAspectRatio: aspectRatio,
    48	      );
    49	    } on FirebaseFunctionsException catch (error) {
    50	      throw VeoSceneGenerationException(
    51	        code: error.code,
    52	        message: _messageForCode(error.code, error.message),
    53	      );
    54	    } catch (_) {
    55	      throw const VeoSceneGenerationException(
    56	        code: 'unknown',
    57	        message: 'La génération VEO a échoué pour une raison inconnue.',
    58	      );
    59	    }
    60	  }
    61	
    62	  Future<VeoGenerationJob> checkVeoSceneGeneration({
    63	    required String sceneId,
    64	  }) async {
    65	    try {
    66	      final callable =
    67	          _resolvedFunctions.httpsCallable('checkVeoSceneGeneration');
    68	      final result = await callable.call(<String, dynamic>{'sceneId': sceneId});
    69	      return VeoGenerationJob.fromJson(
    70	        _asMap(result.data),
    71	        fallbackSceneId: sceneId,
    72	      );
    73	    } on FirebaseFunctionsException catch (error) {
    74	      throw VeoSceneGenerationException(
    75	        code: error.code,
    76	        message: _messageForCode(error.code, error.message),
    77	      );
    78	    } catch (_) {
    79	      throw const VeoSceneGenerationException(
    80	        code: 'unknown',
    81	        message: 'Impossible de vérifier l’état de la génération VEO.',
    82	      );
    83	    }
    84	  }
    85	
    86	  Map<String, dynamic> _asMap(dynamic value) {
    87	    if (value is Map<dynamic, dynamic>) {
    88	      return value.map(
    89	        (key, entry) => MapEntry(key.toString(), entry),
    90	      );
    91	    }
    92	    return const <String, dynamic>{};
    93	  }
    94	
    95	  static String _messageForCode(String code, String? fallback) {
    96	    switch (code) {
    97	      case 'unauthenticated':
    98	        return 'Connexion requise pour lancer une génération VEO.';
    99	      case 'permission-denied':
   100	        return 'Accès refusé: le rôle admin est requis pour VEO.';
   101	      case 'unavailable':
   102	        return 'Le backend VEO est temporairement indisponible.';
   103	      case 'internal':
   104	        return 'Le backend VEO a rencontré une erreur interne.';
   105	      case 'deadline-exceeded':
   106	      case 'timeout':
   107	        return 'Le backend VEO a dépassé le délai attendu.';
   108	      case 'unknown':
   109	        return fallback ?? 'Erreur inconnue pendant la génération VEO.';
   110	      default:
   111	        return fallback ?? 'Erreur VEO: $code';
   112	    }
   113	  }
   114	}

    63	  List<Take60SceneMarker> buildTimeline(SceneModel scene) {
    64	    if (scene.markers.isNotEmpty) {
    65	      final markers = [...scene.markers]
    66	        ..sort((left, right) => left.order.compareTo(right.order));
    67	      return markers;
    68	    }
    69	
    70	    final userCharacter = scene.characterToPlay.isEmpty
    71	        ? 'Utilisateur'
    72	        : scene.characterToPlay;
    73	    final aiCharacter = userCharacter.toLowerCase() == 'voyou'
    74	        ? 'Policière'
    75	        : 'Réplique IA';
    76	    final dialogueLines = _extractDialogueLines(scene.dialogueText);
    77	    const durations = <int>[12, 8, 10, 8, 12, 10];
    78	    final labels = <String>[
    79	      'Plan VO3 1 — intro cinéma',
    80	      'Plan utilisateur 1 — réplique',
    81	      'Plan VO3 2 — réaction',
    82	      'Plan utilisateur 2 — réponse',
    83	      'Plan VO3 3 — tension',
    84	      'Plan utilisateur final — dernière réplique',
    85	    ];
    86	    final markerTypes = <GuidedMarkerType>[
    87	      GuidedMarkerType.aiPlan,
    88	      GuidedMarkerType.userPlan,
    89	      GuidedMarkerType.reactionShot,
    90	      GuidedMarkerType.userPlan,
    91	      GuidedMarkerType.aiReply,
    92	      GuidedMarkerType.finalShot,
    93	    ];
    94	    final markers = <Take60SceneMarker>[];
    95	    var start = 0;
    96	
    97	    for (var index = 0; index < durations.length; index++) {
    98	      final duration = durations[index];
    99	      final type = markerTypes[index];
   100	      final requiresUserRecording = type.requiresUserRecording ||
   101	          type == GuidedMarkerType.finalShot;
   102	      markers.add(
   103	        Take60SceneMarker(
   104	          id: 'default_${scene.id}_$index',
   105	          order: index + 1,
   106	          type: type == GuidedMarkerType.finalShot
   107	              ? GuidedMarkerType.userPlan
   108	              : type,
   109	          startSeconds: start,
   110	          endSeconds: start + duration,
   111	          durationSeconds: duration,
   112	          source: requiresUserRecording ? 'user_video' : 'ai_video',
   113	          character: requiresUserRecording ? userCharacter : aiCharacter,
   114	          dialogue: dialogueLines[index % dialogueLines.length],
   115	          cameraPlan: index.isEven ? 'medium_shot' : 'close_up',
   116	          label: labels[index],
   117	          videoUrl: requiresUserRecording
   118	              ? null
   119	              : (scene.videoUrl?.isNotEmpty == true
   120	                  ? scene.videoUrl
   121	                  : _mockAiVideoUrl),
   122	          cueText: requiresUserRecording
   123	              ? 'Joue ta réplique maintenant.'
   124	              : 'Regarde la scène IA.',
   125	        ),
   126	      );
   127	      start += duration;
   128	    }

   168	  Future<Take60UserRecordingDraft> persistRecording({
   169	    required SceneModel scene,
   170	    required Take60SceneMarker marker,
   171	    required String localTempPath,
   172	    required int durationSeconds,
   173	    required UserPlanStatus status,
   174	    String? uploadedVideoUrl,
   175	  }) async {
   176	    final now = DateTime.now();
   177	    String? finalUploadedUrl = uploadedVideoUrl;
   178	
   179	    // Try to upload the captured segment to Storage so that the backend
   180	    // renderer can reach it. On web (no dart:io File) or if upload fails
   181	    // we silently fall back to the local path — the UI still works in
   182	    // demo / offline mode.
   183	    if (finalUploadedUrl == null &&
   184	        currentUserId != 'guest' &&
   185	        !kIsWeb &&
   186	        localTempPath.isNotEmpty &&
   187	        !localTempPath.startsWith('http')) {
   188	      try {
   189	        final file = File(localTempPath);
   190	        if (await file.exists()) {
   191	          finalUploadedUrl = await _storage.uploadGuidedSegment(
   192	            uid: currentUserId,
   193	            sceneId: scene.id,
   194	            markerId: marker.id,
   195	            file: file,
   196	          );
   197	        }
   198	      } catch (_) {
   199	        // Upload failure is non-fatal — keep local path.
   200	      }
   201	    }
   202	
   203	    final recording = Take60UserRecordingDraft(
   204	      recordingId: _uuid.v4(),
   205	      sceneId: scene.id,
   206	      userId: currentUserId,
   207	      markerId: marker.id,
   208	      localTempPath: localTempPath,
   209	      uploadedVideoUrl: finalUploadedUrl,
   210	      durationSeconds: durationSeconds,
   211	      status: status,
   212	      createdAt: now,
   213	      updatedAt: now,
   214	    );
   215	
   216	    if (currentUserId != 'guest') {
   217	      await _firestore
   218	          .collection('take60_user_recordings')
   219	          .doc(recording.recordingId)
   220	          .set(recording.toMap(), SetOptions(merge: true))

   227	  Future<Take60RenderResult> renderTake60GuidedScene({
   228	    required SceneModel scene,
   229	    required List<Take60UserRecordingDraft> recordings,
   230	  }) async {
   231	    final markers = buildTimeline(scene);
   232	    final payload = {
   233	      'sceneId': scene.id,
   234	      'userId': currentUserId,
   235	      'aiSegments': markers
   236	          .where((marker) => !marker.requiresUserRecording)
   237	          .map(
   238	            (marker) => {
   239	              'markerId': marker.id,
   240	              'type': marker.type.value,
   241	              'videoUrl': marker.videoUrl ?? scene.videoUrl,
   242	              'durationSeconds': marker.durationSeconds,
   243	              'order': marker.order,
   244	            },
   245	          )
   246	          .toList(),
   247	      'userSegments': recordings
   248	          .map(
   249	            (recording) => {
   250	              'markerId': recording.markerId,
   251	              'type': 'user_plan',
   252	              'videoUrl': recording.uploadedVideoUrl ?? recording.localTempPath,
   253	              'durationSeconds': recording.durationSeconds,
   254	            },
   255	          )
   256	          .toList(),
   257	      'markers': markers
   258	          .map(
   259	            (marker) => {
   260	              'markerId': marker.id,
   261	              'type': marker.type.value,
   262	              'order': marker.order,
   263	              'durationSeconds': marker.durationSeconds,
   264	            },
   265	          )
   266	          .toList(),
   267	      'audioRules': scene.audioRules.toMap(),
   268	      'maxDurationSeconds': scene.durationSeconds,
   269	    };
   270	
   271	    try {
   272	      final callable = _functions.httpsCallable('renderTake60GuidedScene');
   273	      final result = await callable.call(payload);
   274	      return Take60RenderResult.fromMap(
   275	        Map<String, dynamic>.from(result.data as Map),
   276	      );
   277	    } on FirebaseFunctionsException {
   278	      return _buildLocalRender(scene: scene, recordings: recordings, markers: markers);
   279	    } catch (_) {
   280	      return _buildLocalRender(scene: scene, recordings: recordings, markers: markers);
   281	    }
   282	  }
